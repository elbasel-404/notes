# Neovim Basics

leader + c + m = Mason
leader + f + f = find files (telescope plugin)
leader + c + r = rename symbol
shift + k = documentation for this symbol
leader + s + g = search for a string (grep)
leader + x + x = see problems (trouble plugin)
leader + c + a = code actions
leader + u + (shift + c) = change theme
leader + f + b = switch tabs
[ + b = next tab
] + b = prev tab
leader + | = vertical split
leader + - = horizontal split
ctrl + h = navigate to left split
ctrl + l = navigate to down split
ctrl + j = navigate to down split
ctrl + k = navigate to up split
ctrl + arrow keys = control split size
leader + s + h = view help pages
leader + s + t = browse todos
ctrl + n = scroll down in autocompletion popup
leader + s + k = search keymaps
leader + s + r = search and replace
leader + g + g = lazygit
leader + f + t = terminal
leader + u + l = toggle line numbers
